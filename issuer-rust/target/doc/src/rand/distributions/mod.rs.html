11//! Sampling from random distributions.
12//!
13//! This is a generalization of `Rand` to allow parameters to control the
14//! exact properties of the generated values, e.g. the mean and standard
15//! deviation of a normal distribution. The `Sample` trait is the most
16//! general, and allows for generating values that change some state
17//! internally. The `IndependentSample` trait is for generating values
18//! that do not need to record state.
19

43/// Types that can be used to create a random instance of `Support`.
44pub trait Sample<Support> {
45    /// Generate a random value of `Support`, using `rng` as the
46    /// source of randomness.
47    fn sample<R: Rng>(&mut self, rng: &mut R) -> Support;
48}
49
50/// `Sample`s that do not require keeping track of state.
51///
52/// Since no state is recorded, each sample is (statistically)
53/// independent of all others, assuming the `Rng` used has this
54/// property.
55// FIXME maybe having this separate is overkill (the only reason is to
56// take &self rather than &mut self)? or maybe this should be the
57// trait called `Sample` and the other should be `DependentSample`.
58pub trait IndependentSample<Support>: Sample<Support> {
59    /// Generate a random value.
60    fn ind_sample<R: Rng>(&self, &mut R) -> Support;
61}

63/// A wrapper for generating types that implement `Rand` via the
64/// `Sample` & `IndependentSample` traits.
65#[derive(Debug)]
66pub struct RandSample<Sup> {
67    _marker: marker::PhantomData<fn() -> Sup>,
68}
69
70impl<Sup> Copy for RandSample<Sup> {}
71impl<Sup> Clone for RandSample<Sup> {
72    fn clone(&self) -> Self { *self }
73}
74
75impl<Sup: Rand> Sample<Sup> for RandSample<Sup> {
76    fn sample<R: Rng>(&mut self, rng: &mut R) -> Sup { self.ind_sample(rng) }
77}
78
79impl<Sup: Rand> IndependentSample<Sup> for RandSample<Sup> {
80    fn ind_sample<R: Rng>(&self, rng: &mut R) -> Sup {
81        rng.gen()
82    }
83}
84
85impl<Sup> RandSample<Sup> {
86    pub fn new() -> RandSample<Sup> {
87        RandSample { _marker: marker::PhantomData }
88    }
89}

91/// A value with a particular weight for use with `WeightedChoice`.
92#[derive(Copy, Clone, Debug)]
93pub struct Weighted<T> {
94    /// The numerical weight of this item
95    pub weight: u32,
96    /// The actual item which is being weighted
97    pub item: T,
98}
99
100/// A distribution that selects from a finite collection of weighted items.
101///
102/// Each item has an associated weight that influences how likely it
103/// is to be chosen: higher weight is more likely.
104///
105/// The `Clone` restriction is a limitation of the `Sample` and
106/// `IndependentSample` traits. Note that `&T` is (cheaply) `Clone` for
107/// all `T`, as is `u32`, so one can store references or indices into
108/// another vector.
109///
110/// # Example
111///
112/// ```rust
113/// use rand::distributions::{Weighted, WeightedChoice, IndependentSample};
114///
115/// let mut items = vec!(Weighted { weight: 2, item: 'a' },
116///                      Weighted { weight: 4, item: 'b' },
117///                      Weighted { weight: 1, item: 'c' });
118/// let wc = WeightedChoice::new(&mut items);
119/// let mut rng = rand::thread_rng();
120/// for _ in 0..16 {
121///      // on average prints 'a' 4 times, 'b' 8 and 'c' twice.
122///      println!("{}", wc.ind_sample(&mut rng));
123/// }
124/// ```
125#[derive(Debug)]
126pub struct WeightedChoice<'a, T:'a> {
127    items: &'a mut [Weighted<T>],
128    weight_range: Range<u32>
129}
130
131impl<'a, T: Clone> WeightedChoice<'a, T> {
132    /// Create a new `WeightedChoice`.
133    ///
134    /// Panics if:
135    ///
136    /// - `items` is empty
137    /// - the total weight is 0
138    /// - the total weight is larger than a `u32` can contain.
139    pub fn new(items: &'a mut [Weighted<T>]) -> WeightedChoice<'a, T> {
140        // strictly speaking, this is subsumed by the total weight == 0 case
141        assert!(!items.is_empty(), "WeightedChoice::new called with no items");
142
143        let mut running_total: u32 = 0;
144
145        // we convert the list from individual weights to cumulative
146        // weights so we can binary search. This *could* drop elements
147        // with weight == 0 as an optimisation.
148        for item in items.iter_mut() {
149            running_total = match running_total.checked_add(item.weight) {
150                Some(n) => n,
151                None => panic!("WeightedChoice::new called with a total weight \
152                               larger than a u32 can contain")
153            };
154
155            item.weight = running_total;
156        }
157        assert!(running_total != 0, "WeightedChoice::new called with a total weight of 0");
158
159        WeightedChoice {
160            items: items,
161            // we're likely to be generating numbers in this range
162            // relatively often, so might as well cache it
163            weight_range: Range::new(0, running_total)
164        }
165    }
166}
167
168impl<'a, T: Clone> Sample<T> for WeightedChoice<'a, T> {
169    fn sample<R: Rng>(&mut self, rng: &mut R) -> T { self.ind_sample(rng) }
170}

172impl<'a, T: Clone> IndependentSample<T> for WeightedChoice<'a, T> {
173    fn ind_sample<R: Rng>(&self, rng: &mut R) -> T {
174        // we want to find the first element that has cumulative
175        // weight > sample_weight, which we do by binary since the
176        // cumulative weights of self.items are sorted.
177
178        // choose a weight in [0, total_weight)
179        let sample_weight = self.weight_range.ind_sample(rng);
180
181        // short circuit when it's the first item
182        if sample_weight < self.items[0].weight {
183            return self.items[0].item.clone();
184        }
185
186        let mut idx = 0;
187        let mut modifier = self.items.len();
188
189        // now we know that every possibility has an element to the
190        // left, so we can just search for the last element that has
191        // cumulative weight <= sample_weight, then the next one will
192        // be "it". (Note that this greatest element will never be the
193        // last element of the vector, since sample_weight is chosen
194        // in [0, total_weight) and the cumulative weight of the last
195        // one is exactly the total weight.)
196        while modifier > 1 {
197            let i = idx + modifier / 2;
198            if self.items[i].weight <= sample_weight {
199                // we're small, so look to the right, but allow this
200                // exact element still.
201                idx = i;
202                // we need the `/ 2` to round up otherwise we'll drop
203                // the trailing elements when `modifier` is odd.
204                modifier += 1;
205            } else {
206                // otherwise we're too big, so go left. (i.e. do
207                // nothing)
208            }
209            modifier /= 2;
210        }
211        return self.items[idx + 1].item.clone();
212    }
213}

215/// Sample a random number using the Ziggurat method (specifically the
216/// ZIGNOR variant from Doornik 2005). Most of the arguments are
217/// directly from the paper:
218///
219/// * `rng`: source of randomness
220/// * `symmetric`: whether this is a symmetric distribution, or one-sided with P(x < 0) = 0.
221/// * `X`: the $x_i$ abscissae.
222/// * `F`: precomputed values of the PDF at the $x_i$, (i.e. $f(x_i)$)
223/// * `F_DIFF`: precomputed values of $f(x_i) - f(x_{i+1})$
224/// * `pdf`: the probability density function
225/// * `zero_case`: manual sampling from the tail when we chose the
226///    bottom box (i.e. i == 0)
227
228// the perf improvement (25-50%) is definitely worth the extra code
229// size from force-inlining.
230#[cfg(feature="std")]
231#[inline(always)]
232fn ziggurat<R: Rng, P, Z>(
233            rng: &mut R,
234            symmetric: bool,
235            x_tab: ziggurat_tables::ZigTable,
236            f_tab: ziggurat_tables::ZigTable,
237            mut pdf: P,
238            mut zero_case: Z)
239            -> f64 where P: FnMut(f64) -> f64, Z: FnMut(&mut R, f64) -> f64 {
240    const SCALE: f64 = (1u64 << 53) as f64;
241    loop {
242        // reimplement the f64 generation as an optimisation suggested
243        // by the Doornik paper: we have a lot of precision-space
244        // (i.e. there are 11 bits of the 64 of a u64 to use after
245        // creating a f64), so we might as well reuse some to save
246        // generating a whole extra random number. (Seems to be 15%
247        // faster.)
248        //
249        // This unfortunately misses out on the benefits of direct
250        // floating point generation if an RNG like dSMFT is
251        // used. (That is, such RNGs create floats directly, highly
252        // efficiently and overload next_f32/f64, so by not calling it
253        // this may be slower than it would be otherwise.)
254        // FIXME: investigate/optimise for the above.
255        let bits: u64 = rng.gen();
256        let i = (bits & 0xff) as usize;
257        let f = (bits >> 11) as f64 / SCALE;
258
259        // u is either U(-1, 1) or U(0, 1) depending on if this is a
260        // symmetric distribution or not.
261        let u = if symmetric {2.0 * f - 1.0} else {f};
262        let x = u * x_tab[i];
263
264        let test_x = if symmetric { x.abs() } else {x};
265
266        // algebraically equivalent to |u| < x_tab[i+1]/x_tab[i] (or u < x_tab[i+1]/x_tab[i])
267        if test_x < x_tab[i + 1] {
268            return x;
269        }
270        if i == 0 {
271            return zero_case(rng, u);
272        }
273        // algebraically equivalent to f1 + DRanU()*(f0 - f1) < 1
274        if f_tab[i + 1] + (f_tab[i] - f_tab[i + 1]) * rng.gen::<f64>() < pdf(x) {
275            return x;
276        }
277    }
278}
